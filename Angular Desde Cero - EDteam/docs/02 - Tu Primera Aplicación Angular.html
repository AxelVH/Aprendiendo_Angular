Generación de codigo con Angular CLI


        

            Generación de código
            ng generate < contenido > [opciones]

            //generar una clase typescript
            ng generate class < nombre > [opciones]
            ng generate class curso

            //generar interfaces
            ng generate interface < nombre > [opciones]
            ng generate interface escuela

            //generar enumerados
            ng generate enum < nombre > [opciones]
            ng generate enum dia

            //generar componentes
            ng generate component < nombre > [opciones]
            ng generate component escuela-digital

            //generar modulo
            ng generate module < nombre > [opciones]
            ng generate module escuela-digital

            //generar servicios
            ng generate service < nombre > [opciones]
            ng generate service escuela-digital

            //generar directivas
            ng generate directive < nombre > [opciones]
            ng generate directive escuela-digital

            //generate pipes
            ng generate pipe < nombre > [opciones]
            ng generate pipe filtro

            //otros
            appShell
            application
            guard
            library
            serviceWorker
            universal
            webWorker

            //fuente
            angular.io/cli/generate
